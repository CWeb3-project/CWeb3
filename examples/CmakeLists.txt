cmake_minimum_required(VERSION 3.9)
project("server_test" VERSION 1.0.1 DESCRIPTION "simple server test")

set(SRCDir "${CMAKE_SOURCE_DIR}/")
set(HDRDir "${CMAKE_SOURCE_DIR}/include")

file (GLOB SRCFiles ${SRCDir}/webserver.c)
file (GLOB HDRFiles ${HDRDir}/*.h)

add_executable(${PROJECT_NAME}
    ${SRCFiles} ${HDRFiles}
)
target_link_libraries(${PROJECT_NAME} "${CMAKE_SOURCE_DIR}/libCWeb3.so")
